## 1.准备文件
```shell
HJB.noN.vcf.gz
```
## 2.计算R2
```shell
## 做PopLDdecay每个群体至少要三个个体
for i in `cat /home/caominghao/01.data/02.cmh/01.HJB/05.02.233pop_diversi/04.LDdecay/list`;
do
time PopLDdecay -InVCF /home/caominghao/01.data/02.cmh/01.HJB/05.02.233pop_diversi/04.LDdecay/00.vcf/HJB.noN.vcf.gz -OutStat HJB.noN.${i} -SubPop ${i}.list &
done
```
## 3.绘图
```shell
# 生成清单
## 生成文件路径
ls -d /home/caominghao/01.data/02.cmh/01.HJB/05.02.233pop_diversi/04.LDdecay/*.stat.gz >> inFile.list
## 生成群体名字
for i in *.list;do echo -e "`basename ${i} .list`" >> list;done
## 以列方式合并两文件
paste -d ' ' inFile.list list > stat.list
# 绘图
time perl /home/caominghao/Data/01.cmh/software/PopLDdecay/PopLDdecay-master/bin/Plot_MultiPop.pl -inList stat.list -output HJB
```
## 4.提取LD衰减距离
```shell
# 给每一个群体都提取一个LD衰减距离，r^2值是根据LD最大值一半时的距离，四舍五入的原则。
zcat HJB.new.Clade1.stat.gz | awk '($2 == 0.376) {print $0}' | less -SN
zcat HJB.new.Clade2.stat.gz | awk '($2 == 0.343) {print $0}' | less -SN
zcat HJB.new.Clade3.stat.gz | awk '($2 == 0.358) {print $0}' | less -SN
zcat HJB.new.Clade4.stat.gz | awk '($2 == 0.386) {print $0}' | less -SN
zcat HJB.new.Clade5.stat.gz | awk '($2 == 0.344) {print $0}' | less -SN
zcat HJB.new.Clade6.stat.gz | awk '($2 == 0.337) {print $0}' | less -SN
zcat HJB.new.Clade7.stat.gz | awk '($2 == 0.418) {print $0}' | less -SN
```
